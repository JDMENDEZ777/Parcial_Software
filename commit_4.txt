DROP MATERIALIZED VIEW IF EXISTS mv_stock_por_tienda;
CREATE MATERIALIZED VIEW mv_stock_por_tienda AS
SELECT tp.id_tienda,
       t.nombre_tienda,
       p.id_producto,
       p.cod_producto,
       p.descripcion,
       tp.cantidad,
       tp.saldo,
       tp.fecha_actualizacion
FROM tienda_producto tp
JOIN tienda t ON t.id_tienda = tp.id_tienda
JOIN producto p ON p.id_producto = tp.id_producto;

CREATE INDEX IF NOT EXISTS ix_mv_stock_tienda_producto ON mv_stock_por_tienda(id_tienda, id_producto);
